Un dégradé linéaire

div {
background: -webkit-gradient(linear, left top, right top, from(red), to(yellow));
background: linear-gradient(0deg, yellow, red);
}

div {
background: -webkit-gradient(linear, left bottom, right top, from(red), to(yellow));
background: linear-gradient(45deg, yellow, red);
}

div {
background: -webkit-gradient(linear, left bottom, right bottom, color-stop(0.25, red), color-stop(0.75, yellow));
background: linear-gradient(left, yellow 25%, red 75%);
}

div {
background: -webkit-gradient(linear, left bottom, right bottom, from(red), color-stop(0.70, orange), to(yellow));
background: linear-gradient(left, yellow, orange 75%, red);
}

Un dégradé radial

div {
background: -webkit-gradient(radial, 50% 50%, 0, 50% 50%, 212, from(red), to(yellow));
background: radial-gradient(red, yellow);
}

Webkit

}

div {
background: -webkit-gradient(radial, 150 10, 20, 150 200, 50, from(white), to(silver)), silver;
}
